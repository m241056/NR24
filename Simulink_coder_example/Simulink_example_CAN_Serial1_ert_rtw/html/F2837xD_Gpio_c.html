19	//Sets all pins to be muxed to GPIO in input mode with pull-ups enabled.



20	//Also resets CPU control to CPU1 and disables open drain and polarity



21	//inversion and sets the qualification to synchronous. Also unlocks all GPIOs



22	//Only one CPU should call this function.



23	void InitGpio()



24	{



25		volatile Uint32 *gpioBaseAddr;



26		Uint16 regOffset;



27



28		//Disable pin locks



29		EALLOW;



30		GpioCtrlRegs.GPALOCK.all = 0x00000000;



31		GpioCtrlRegs.GPBLOCK.all = 0x00000000;



32		GpioCtrlRegs.GPCLOCK.all = 0x00000000;



33		GpioCtrlRegs.GPDLOCK.all = 0x00000000;



34		GpioCtrlRegs.GPELOCK.all = 0x00000000;



35		GpioCtrlRegs.GPFLOCK.all = 0x00000000;



36



37		//Fill all registers with zeros. Writing to each register separately



38		//for six GPIO modules would make this function *very* long. Fortunately,



39		//we'd be writing them all with zeros anyway, so this saves a lot of space.



40		gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;



41		for (regOffset = 0; regOffset < sizeof(GpioCtrlRegs)/2; regOffset++)



42		{



43			//Hack to avoid enabling pull-ups on all pins. GPyPUD is offset



44			//0x0C in each register group of 0x40 words. Since this is a



45			//32-bit pointer, the addresses must be divided by 2.



46			if (regOffset % (0x40/2) != (0x0C/2))



47				gpioBaseAddr[regOffset] = 0x00000000;



48		}



49



50		gpioBaseAddr = (Uint32 *)&GpioDataRegs;



51		for (regOffset = 0; regOffset < sizeof(GpioDataRegs)/2; regOffset++)



52		{



53			gpioBaseAddr[regOffset] = 0x00000000;



54		}



55



56		EDIS;



57	}

60	//Set the peripheral muxing for the specified pin. The appropriate



61    //parameters can be found in the GPIO Muxed Pins table(4.4) in the datasheet.



62    //Use the GPIO index row (0 to 15) to select a muxing option for the GPIO.



63	void GPIO_SetupPinMux(Uint16 pin, Uint16 cpu, Uint16 peripheral)



64	{



65		volatile Uint32 *gpioBaseAddr;



66		volatile Uint32 *mux, *gmux, *csel;



67		Uint16 pin32, pin16, pin8;



68



69		pin32 = pin % 32;



70		pin16 = pin % 16;



71		pin8 = pin % 8;



72		gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;



73



74		//Sanity check for valid cpu and peripheral values



75		if (cpu > GPIO_MUX_CPU2CLA || peripheral > 0xF)



76			return;



77



78		//Create pointers to the appropriate registers. This is a workaround



79		//for the way GPIO registers are defined. The standard definition



80		//in the header file makes it very easy to do named accesses of one



81		//register or bit, but hard to do arbitrary numerical accesses. It's



82		//easier to have an array of GPIO modules with identical registers,



83		//including arrays for multi-register groups like GPyCSEL1-4. But



84		//the header file doesn't define anything we can turn into an array,



85		//so manual pointer arithmetic is used instead.



86		mux = gpioBaseAddr + GPYMUX + pin32/16;



87		gmux = gpioBaseAddr + GPYGMUX + pin32/16;



88		csel = gpioBaseAddr + GPYCSEL + pin32/8;



89



90		//Now for the actual function



91		EALLOW;



92



93		//To change the muxing, set the peripheral mux to 0/GPIO first to avoid



94		//glitches, then change the group mux, then set the peripheral mux to



95		//its target value. Finally, set the CPU select. This procedure is



96		//described in the TRM. Unfortunately, since we don't know the pin in



97		//advance we can't hardcode a bitfield reference, so there's some tricky



98		//bit twiddling here.



99		*mux &= ~(0x3UL << (2*pin16));



100		*gmux &= ~(0x3UL << (2*pin16));



101		*gmux |= (Uint32)((peripheral >> 2) & 0x3UL) << (2*pin16);



102		*mux |= (Uint32)(peripheral & 0x3UL) << (2*pin16);



103



104		*csel &= ~(0x3L << (4*pin8));



105		*csel |= (Uint32)(cpu & 0x3L) << (4*pin8);



106



107		//WARNING: This code does not touch the analog mode select registers,



108		//which are needed to give the USB module control of its IOs.



109



110		EDIS;



111	}

114	//Setup up the GPIO input/output options for the specified pin.



115	//The flags are a 16-bit mask produced by ORing together options.



116	//For input pins, the valid flags are:



117	//GPIO_PULLUP	Enable pull-up



118	//GPIO_INVERT	Enable input polarity inversion



119	//GPIO_SYNC		Synchronize the input latch to PLLSYSCLK (default -- you don't need to specify this)



120	//GPIO_QUAL3	Use 3-sample qualification



121	//GPIO_QUAL6	Use 6-sample qualification



122	//GPIO_ASYNC	Do not use synchronization or qualification



123	//(Note: only one of SYNC, QUAL3, QUAL6, or ASYNC is allowed)



124	//



125	//For output pins, the valid flags are:



126	//GPIO_OPENDRAIN	Output in open drain mode



127	//GPIO_PULLUP		If open drain enabled, also enable the pull-up



128	//and the input qualification flags (SYNC/QUAL3/QUAL6/SYNC) listed above.



129	//



130	//With no flags, the default input state is synchronous with no



131	//pull-up or polarity inversion. The default output state is



132	//the standard digital output.



133	void GPIO_SetupPinOptions(Uint16 pin, Uint16 output, Uint16 flags)



134	{



135		volatile Uint32 *gpioBaseAddr;



136		volatile Uint32 *dir, *pud, *inv, *odr, *qsel;



137		Uint32 pin32, pin16, pinMask, qual;



138



139		pin32 = pin % 32;



140		pin16 = pin % 16;



141		pinMask = 1UL << pin32;



142		gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;



143



144		//Create pointers to the appropriate registers. This is a workaround



145		//for the way GPIO registers are defined. The standard definition



146		//in the header file makes it very easy to do named accesses of one



147		//register or bit, but hard to do arbitrary numerical accesses. It's



148		//easier to have an array of GPIO modules with identical registers,



149		//including arrays for multi-register groups like GPyQSEL1-2. But



150		//the header file doesn't define anything we can turn into an array,



151		//so manual pointer arithmetic is used instead.



152		dir = gpioBaseAddr + GPYDIR;



153		pud = gpioBaseAddr + GPYPUD;



154		inv = gpioBaseAddr + GPYINV;



155		odr = gpioBaseAddr + GPYODR;



156		qsel = gpioBaseAddr + GPYQSEL + pin32/16;



157



158		EALLOW;



159



160		//Set the data direction



161		*dir &= ~pinMask;



162		if (output == 1)



163		{



164			//Output, with optional open drain mode and pull-up



165			*dir |= pinMask;



166



167			//Enable open drain if necessary



168			if (flags & GPIO_OPENDRAIN)



169				*odr |= pinMask;



170			else



171				*odr &= ~pinMask;



172



173			//Enable pull-up if necessary. Open drain mode must be active.



174			if (flags & (GPIO_OPENDRAIN | GPIO_PULLUP))



175				*pud &= ~pinMask;



176			else



177				*pud |= pinMask;



178		} else



179		{



180			//Input, with optional pull-up, qualification, and polarity inversion



181			*dir &= ~pinMask;



182



183			//Enable pull-up if necessary



184			if (flags & GPIO_PULLUP)



185				*pud &= ~pinMask;



186			else



187				*pud |= pinMask;



188



189			//Invert polarity if necessary



190			if (flags & GPIO_INVERT)



191				*inv |= pinMask;



192			else



193				*inv &= ~pinMask;



194		}



195



196		//Extract the qualification parameter and load it into the register. This is



197		//also needed for open drain outputs, so we might as well do it all the time.



198		qual = (flags & GPIO_ASYNC) / GPIO_QUAL3;



199		*qsel &= ~(0x3L << (2 * pin16));



200		if (qual != 0x0)



201			*qsel |= qual << (2 * pin16);



202



203		EDIS;



204	}

206	//Enable or disable the GPIO register bit lock for the specified pin.



207	//The valid flags are:



208	//GPIO_UNLOCK	Unlock the pin setup register bits for the specified pin



209	//GPIO_LOCK 	Lock the pin setup register bits for the specified pin



210	void GPIO_SetupLock(Uint16 pin, Uint16 flags)



211	{



212		volatile Uint32 *gpioBaseAddr;



213		volatile Uint32 *lock;



214		Uint32 pin32, pinMask;



215



216		pin32 = pin % 32;



217		pinMask = 1UL << pin32;



218		gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;



219



220		//Create pointers to the appropriate registers. This is a workaround



221		//for the way GPIO registers are defined. The standard definition



222		//in the header file makes it very easy to do named accesses of one



223		//register or bit, but hard to do arbitrary numerical accesses. It's



224		//easier to have an array of GPIO modules with identical registers,



225		//including arrays for multi-register groups like GPyQSEL1-2. But



226		//the header file doesn't define anything we can turn into an array,



227		//so manual pointer arithmetic is used instead.



228		lock = gpioBaseAddr + GPYLOCK;



229



230



231		EALLOW;



232		if(flags){



233			//Lock the pin



234			*lock |= pinMask;



235		}else{



236			//Unlock the pin



237			*lock &= ~pinMask;



238		}



239		EDIS;



240	}

348//Read the GPyDAT register bit for the specified pin. Note that this returns



349//the actual state of the pin, not the state of the output latch.



350Uint16 GPIO_ReadPin(Uint16 pin)



351{



352	volatile Uint32 *gpioDataReg;



353	Uint16 pinVal;



354



355	gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (pin/32)*GPY_DATA_OFFSET;



356	pinVal = (gpioDataReg[GPYDAT] >> (pin % 32)) & 0x1;



357



358	return pinVal;



359}

361//Set the GPyDAT register bit for the specified pin.



362void GPIO_WritePin(Uint16 pin, Uint16 outVal)



363{



364	volatile Uint32 *gpioDataReg;



365	Uint32 pinMask;



366



367	gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (pin/32)*GPY_DATA_OFFSET;



368	pinMask = 1UL << (pin % 32);



369



370	if (outVal == 0)



371		gpioDataReg[GPYCLEAR] = pinMask;



372	else



373		gpioDataReg[GPYSET] = pinMask;



374}
