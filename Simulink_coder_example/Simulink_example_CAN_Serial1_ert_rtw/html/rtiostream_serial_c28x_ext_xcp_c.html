89interrupt void SCIRXINT_isr(void)



90{



91    if(queRx.queCount < QUE_SIZE )



92        queRx.queElements[((queRx.firstElement + queRx.queCount++) & (QUE_SIZE-1))] = MW_SCI_PTR->SCIRXBUF.all;



93    else



94        while(1){}



95    



96  /*



97  	 Assumption: queRx (SW Queue) will not over flow if the QUE_SIZE is high (64 default value)



98	 The queRx is used to collect data when the SCI HW FIFO overflows



99	 Otherwise the Data Collection for external Mode occurs by



100	 polling the SCI Rx Buffer



101  */



102



103  MW_SCI_PTR->SCIFFRX.bit.RXFFINTCLR = 1; /* Clear Interrupt flag*/



104  #if MW_EXTMODE_SCIMODULE == 0 || MW_EXTMODE_SCIMODULE == 1



105  PieCtrlRegs.PIEACK.all = PIEACK_GROUP9;/* Acknowledge to receive more interrupts   */  



106  #elif MW_EXTMODE_SCIMODULE == 2 || MW_EXTMODE_SCIMODULE == 3



107  PieCtrlRegs.PIEACK.all = PIEACK_GROUP8;/* Acknowledge to receive more interrupts   */ 



108  #endif



109}

111void enable_interrupts_sci(void)



112{



113    #if MW_EXTMODE_SCIMODULE == 0



114    EALLOW;



115      #ifdef F2837X_REG_FORMAT



116      PieVectTable.SCIA_RX_INT = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



117      #else



118      PieVectTable.SCIRXINTA = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



119      #endif //F2837X_REG_FORMAT



120    PieCtrlRegs.PIEIER9.bit.INTx1 = 1;   /* Enable PIE group 9 interrupt 1 for SCIRXINTA*/



121    EDIS;



122    IER |= M_INT9;



123    #elif MW_EXTMODE_SCIMODULE == 1



124    EALLOW;



125      #ifdef F2837X_REG_FORMAT



126      PieVectTable.SCIB_RX_INT = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



127      #else



128      PieVectTable.SCIRXINTB = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



129      #endif //F2837X_REG_FORMAT



130    PieCtrlRegs.PIEIER9.bit.INTx3 = 1;   /* Enable PIE group 9 interrupt 3 for SCIRXINTB*/



131    EDIS;



132    IER |= M_INT9;



133    #elif MW_EXTMODE_SCIMODULE == 2



134    EALLOW;



135      #ifdef F2837X_REG_FORMAT



136      PieVectTable.SCIC_RX_INT = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



137      #else



138      PieVectTable.SCIRXINTC = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



139      #endif //F2837X_REG_FORMAT



140    PieCtrlRegs.PIEIER8.bit.INTx5 = 1;   /* Enable PIE group 8 interrupt 5 for SCIRXINTC*/



141    EDIS;



142    IER |= M_INT8;



143    #elif MW_EXTMODE_SCIMODULE == 3



144    EALLOW;



145      #ifdef F2837X_REG_FORMAT



146      PieVectTable.SCID_RX_INT = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



147      #else



148      PieVectTable.SCIRXINTD = &SCIRXINT_isr;/* Hook interrupt to the ISR*/



149      #endif //F2837X_REG_FORMAT



150    PieCtrlRegs.PIEIER8.bit.INTx7 = 1;   /* Enable PIE group 8 interrupt 7 for SCIRXINTD*/



151    EDIS;



152    IER |= M_INT8;



153    #endif



154}

157int rtIOStreamOpen(int argc, void * argv[])



158{



159    int streamID = SERVER_STREAM_ID;



160    init_SCI_external_mode();



161    Set_GPIOPin(EXT_SCI_RX_PIN, GPIO_MUX_CPU1, RX_PIN_MUX_VAL, GPIO_INPUT, 0);



162    Set_GPIOPin(EXT_SCI_TX_PIN, GPIO_MUX_CPU1, TX_PIN_MUX_VAL, GPIO_OUTPUT, 0);



163    initQue(&queRx);				/* int SW Que that supplements SCI for External Mode Over Serial */



164    enable_interrupts_sci();		/* Enable SCI interrupt used for external mode over Serial*/



165



166    return streamID;



167}

177/**



178 * @brief   Send BYTEs over UART



179 * 



180 * @note    Send all octets in a word, LSB first.



181 *



182 * @param[in]   streamID    



183 * @param[in]   src         Source buffer



184 * @param[in]   bytesToSend Number of BYTEs to send



185 * @param[out]  bytesSent   Number of BYTEs sent



186 */



187int rtIOStreamSend(int streamID, const void *src, size_t bytesToSend, size_t *bytesSent)



188{



189    sci_xmit_external_mode((char *)src, bytesToSend);



190



191    *bytesSent = bytesToSend;



192    return RTIOSTREAM_NO_ERROR;



193}

195/**



196 * @brief   Receive BYTEs from UART



197 * 



198 * @note    Pack all received BYTEs into memory, LSB first



199 *



200 * @param[in]   streamID    



201 * @param[in]   dst            Destination buffer



202 * @param[in]   bytesToReceive Number of BYTEs to receive



203 * @param[out]  bytesRcvd      Number of BYTEs received



204 */



205int rtIOStreamRecv(int streamID, void *dst, size_t bytesToReceive, size_t *bytesRcvd)



206{



207	int errFlg = NOERROR;



208



209	*bytesRcvd = 0;



210



211	errFlg = sci_rcv_external_mode((char *)dst, bytesToReceive, (int *)bytesRcvd);



212



213	if (errFlg != NOERROR)



214		return RTIOSTREAM_ERROR;



215



216	return RTIOSTREAM_NO_ERROR;



217}

222void init_SCI_external_mode(void)



223{



224    #if MW_EXTMODE_SCIMODULE == 0



225    MW_SCI_PTR = &SciaRegs;



226    #elif MW_EXTMODE_SCIMODULE == 1



227    MW_SCI_PTR = &ScibRegs;



228    #elif MW_EXTMODE_SCIMODULE == 2



229    MW_SCI_PTR = &ScicRegs;



230    #elif MW_EXTMODE_SCIMODULE == 3



231    MW_SCI_PTR = &ScidRegs;



232    #endif



233	EALLOW;



234    /* initialize SCI & FIFO registers */



235	/*



236	* Initialize SCI with following parameters:



237	*    BaudRate              : 115200



238	*    CharacterLengthBits   : 8



239	*    EnableLoopBack        : 0



240	*    NumberOfStopBits      : 1



241	*    ParityMode            : None



242	*    SuspensionMode        : Free_run



243	*    CommMode              : Raw_data



244	*/



245    #ifdef F2837X_REG_FORMAT



246        #if MW_EXTMODE_SCIMODULE == 0



247        CpuSysRegs.PCLKCR7.bit.SCI_A = 1;



248        #elif MW_EXTMODE_SCIMODULE == 1



249        CpuSysRegs.PCLKCR7.bit.SCI_B = 1;



250        #elif MW_EXTMODE_SCIMODULE == 2



251        CpuSysRegs.PCLKCR7.bit.SCI_C = 1;



252        #elif MW_EXTMODE_SCIMODULE == 3



253        CpuSysRegs.PCLKCR7.bit.SCI_D = 1;



254        #endif



255    #endif



256	MW_SCI_PTR->SCICTL1.bit.SWRESET = 0;    //Software reset



257	MW_SCI_PTR->SCICTL1.bit.SWRESET = 1;    //Software reset



258	MW_SCI_PTR->SCICCR.bit.STOPBITS = 0;    //Number of stop bits. (0: One stop bit, 1: Two stop bits)



259	MW_SCI_PTR->SCICCR.bit.PARITY = 0;      //Parity mode (0: Odd parity, 1: Even parity)



260	MW_SCI_PTR->SCICCR.bit.PARITYENA = 0;   //Enable Parity Mode



261	MW_SCI_PTR->SCICCR.bit.LOOPBKENA = 0;   //Loop Back enable



262	MW_SCI_PTR->SCICCR.bit.ADDRIDLE_MODE = 0;//ADDR/IDLE Mode control



263	MW_SCI_PTR->SCICCR.bit.SCICHAR = 7;     //Character length



264	MW_SCI_PTR->SCICTL1.bit.RXERRINTENA = 0;//Disable receive error interrupt



265	MW_SCI_PTR->SCICTL1.bit.TXENA = 1;      // SCI transmitter enable



266	MW_SCI_PTR->SCICTL1.bit.RXENA = 1;      // SCI receiver enable



267    



268    #ifdef F2837X_REG_FORMAT



269    MW_SCI_PTR->SCIHBAUD.all = MW_PIL_SCIHBAUD;



270	MW_SCI_PTR->SCILBAUD.all = MW_PIL_SCILBAUD;



271    //Free run, continue SCI operation regardless of suspend



272	MW_SCI_PTR->SCIPRI.bit.FREESOFT = 2;        // Free emulation mode control



273    #else



274	MW_SCI_PTR->SCIHBAUD = MW_PIL_SCIHBAUD;



275	MW_SCI_PTR->SCILBAUD = MW_PIL_SCILBAUD;



276    //Free run, continue SCI operation regardless of suspend



277	MW_SCI_PTR->SCIPRI.bit.FREE = 1;        // Free emulation mode control



278	MW_SCI_PTR->SCIPRI.bit.SOFT = 0;        // Interrupt priority select



279    #endif



280



281	



282	MW_SCI_PTR->SCIFFCT.bit.ABDCLR = 0;



283	MW_SCI_PTR->SCIFFCT.bit.CDC = 0;



284	MW_SCI_PTR->SCIFFTX.bit.SCIRST = 1;     // SCI reset rx/tx channels



285	MW_SCI_PTR->SCIFFTX.bit.SCIFFENA = 1;   // SCI FIFO enhancements are enabled.



286	#ifdef F2837X_REG_FORMAT



287		MW_SCI_PTR->SCIFFTX.bit.TXFIFORESET = 1;// Re-enable transmit FIFO operation.



288	#else



289		MW_SCI_PTR->SCIFFTX.bit.TXFIFOXRESET = 1;// Re-enable transmit FIFO operation.



290	#endif



291	MW_SCI_PTR->SCIFFRX.bit.RXFIFORESET = 1;// Re-enable receive FIFO operation.



292



293



294	MW_SCI_PTR->SCIFFRX.bit.RXFFIENA = 1;					//Receive FIFO Interrupt Enable



295	MW_SCI_PTR->SCIFFRX.bit.RXFFIL = 1;	//Generate Interrupt when FIFO Overflows



296	MW_SCI_PTR->SCICTL1.bit.RXERRINTENA = 1; // Enable receive error interrupt to handle any receive errors for SCI



297	EDIS;



298}

300/**



301 * @brief Receive Data from the SCI module, or from the SW Queue that is filled by the



302 *        SCI module



303 *



304 * @note  This function packs data into the destination buffer.



305 *



306 * @param[out]  rcvBuff          Destination buffer



307 * @param[in]   buffLen          Number of 8bit bytes to read



308 * @param[in]   loopMode         No idea what this does



309 * @param[out]  sizeCharsRcvd    Number of characters read successfully



310 */



311int sci_rcv_external_mode(char *rcvBuff, size_t buffLen, int *sizeCharsRcvd)



312{   



313    uint8_T rcvChar;



314	uint8_T charsRcvd = 0;



315    *sizeCharsRcvd = charsRcvd;



316    



317    if( queRx.queCount < buffLen ){



318        // Add code to flush the HW Buffer and check queCount again before just naively quitting.



319        // If the HW buffer ISRs at 8 bytes, and we have 5 bytes in the buffer, the ISR won't trigger.



320        // rtIOStreamRevc will now pend unnecessarily until we receive 3 more bytes belonging to the next packet.



321        return NOERROR;



322    }



323    



324    /* When an xcp command with odd number of bytes (like one BYTE GET_DAQ_RESOLUTION_INFO/GET_DAQ_PROCESSOR_INFO)



325     * is pushed into packed words, the MSB of the last word will be undetermined.



326     * This value will interfere with the checksum calculation (since the checksum)



327     * works on all bytes of the words.



328     */



329    memset(rcvBuff, 0, (buffLen+1)/2);



330    //__byte(rcvBuff, 1 + (buffLen+1)/2) = 0x00;



331



332	while (charsRcvd < buffLen ) {



333



334		if(deleteElement(&queRx, (char *)&rcvChar))  {		//try to pull off a byte from the queue.



335            __byte((int *)rcvBuff, charsRcvd) = rcvChar ;



336			charsRcvd += 1;



337		}



338



339        #if MW_EXTMODE_SCIMODULE == 0



340		if( (SciaRegs.SCIRXST.bit.FE) || (SciaRegs.SCIRXST.bit.PE) || (SciaRegs.SCIFFRX.bit.RXFFOVF) || (SciaRegs.SCIRXST.bit.BRKDT))



341		{



342        #elif MW_EXTMODE_SCIMODULE == 1



343		if( (ScibRegs.SCIRXST.bit.FE) || (ScibRegs.SCIRXST.bit.PE) || (ScibRegs.SCIFFRX.bit.RXFFOVF) || (ScibRegs.SCIRXST.bit.BRKDT))



344		{



345        #elif MW_EXTMODE_SCIMODULE == 2



346		if( (ScicRegs.SCIRXST.bit.FE) || (ScicRegs.SCIRXST.bit.PE) || (ScicRegs.SCIFFRX.bit.RXFFOVF) || (ScicRegs.SCIRXST.bit.BRKDT))



347		{



348        #elif MW_EXTMODE_SCIMODULE == 3



349		if( (ScidRegs.SCIRXST.bit.FE) || (ScidRegs.SCIRXST.bit.PE) || (ScidRegs.SCIFFRX.bit.RXFFOVF) || (ScidRegs.SCIRXST.bit.BRKDT))



350		{



351        #endif



352			return FRAMERR;					/* Providing FRAME error for frame error, parity error and FIFO overflow



353											 as the RTIOStream doesnt have capability to differentiate between errors */



354		}



355	}



356	*sizeCharsRcvd = charsRcvd;



357	return NOERROR;



358}

360void sci_xmit_external_mode(char* pmsg, int msglen)



361{



362	int i;



363    #if MW_EXTMODE_SCIMODULE == 0



364	for (i = 0; i < msglen; i++) {



365		while (SciaRegs.SCIFFTX.bit.TXFFST == MW_PIL_SCIFIFOLEN) {} /* The buffer is full */



366        #ifdef F2837X_REG_FORMAT



367        SciaRegs.SCITXBUF.bit.TXDT = __byte(pmsg, i);



368        #else



369		SciaRegs.SCITXBUF = __byte(pmsg, i);



370        #endif



371	}



372	while (SciaRegs.SCIFFTX.bit.TXFFST != 0) {}



373    #elif MW_EXTMODE_SCIMODULE == 1



374	for (i = 0; i < msglen; i++) {



375		while (ScibRegs.SCIFFTX.bit.TXFFST == MW_PIL_SCIFIFOLEN) {} /* The buffer is full */



376        #ifdef F2837X_REG_FORMAT



377        ScibRegs.SCITXBUF.bit.TXDT = __byte(pmsg, i);



378        #else



379		ScibRegs.SCITXBUF = __byte(pmsg, i);



380        #endif



381	}



382	while (ScibRegs.SCIFFTX.bit.TXFFST != 0) {}



383    #elif MW_EXTMODE_SCIMODULE == 2



384	for (i = 0; i < msglen; i++) {



385		while (ScicRegs.SCIFFTX.bit.TXFFST == MW_PIL_SCIFIFOLEN) {} /* The buffer is full */



386        #ifdef F2837X_REG_FORMAT



387        ScicRegs.SCITXBUF.bit.TXDT = __byte(pmsg, i);



388        #else



389		ScicRegs.SCITXBUF = __byte(pmsg, i);



390        #endif



391	}



392	while (ScicRegs.SCIFFTX.bit.TXFFST != 0) {}



393    #elif MW_EXTMODE_SCIMODULE == 3



394	for (i = 0; i < msglen; i++) {



395		while (ScidRegs.SCIFFTX.bit.TXFFST == MW_PIL_SCIFIFOLEN) {} /* The buffer is full */



396        #ifdef F2837X_REG_FORMAT



397        ScidRegs.SCITXBUF.bit.TXDT = __byte(pmsg, i);



398        #else



399		ScidRegs.SCITXBUF = __byte(pmsg, i);



400        #endif



401	}



402	while (ScidRegs.SCIFFTX.bit.TXFFST != 0) {}



403    #endif



404



405}
