17// This function initializes the DMA to a known state.



18//



19void DMAInitialize(void)



20{



21    EALLOW;



22



23    // Perform a hard reset on DMA



24    DmaRegs.DMACTRL.bit.HARDRESET = 1;



25   __asm (" nop"); // one NOP required after HARDRESET



26



27    // Allow DMA to run free on emulation suspend



28    DmaRegs.DEBUGCTRL.bit.FREE = 1;



29



30    EDIS;



31}

34void DMACH1AddrConfig(volatile Uint16 *DMA_Dest,volatile Uint16 *DMA_Source)



35{



36    EALLOW;



37    // Set up SOURCE address:



38    DmaRegs.CH1.SRC_BEG_ADDR_SHADOW = (Uint32)DMA_Source;   // Point to beginning of source buffer



39    DmaRegs.CH1.SRC_ADDR_SHADOW =     (Uint32)DMA_Source;



40



41    // Set up DESTINATION address:



42    DmaRegs.CH1.DST_BEG_ADDR_SHADOW = (Uint32)DMA_Dest;     // Point to beginning of destination buffer



43    DmaRegs.CH1.DST_ADDR_SHADOW =     (Uint32)DMA_Dest;



44



45



46    EDIS;



47}

89void DMACH1ModeConfig(Uint16 persel, Uint16 perinte, Uint16 oneshot, Uint16 cont, Uint16 synce, Uint16 syncsel, Uint16 ovrinte, Uint16 datasize, Uint16 chintmode, Uint16 chinte)



90{



91    EALLOW;



92



93    // Set up MODE Register:



94    DmaClaSrcSelRegs.DMACHSRCSEL1.bit.CH1 = persel; // Source select



95    DmaRegs.CH1.MODE.bit.PERINTSEL = 1;             // Should be hard coded to channel, above now selects source



96    DmaRegs.CH1.MODE.bit.PERINTE = perinte;         // Peripheral interrupt enable



97    DmaRegs.CH1.MODE.bit.ONESHOT = oneshot;         // Oneshot enable



98    DmaRegs.CH1.MODE.bit.CONTINUOUS = cont;         // Continuous enable



99    DmaRegs.CH1.MODE.bit.OVRINTE = ovrinte;         // Enable/disable the overflow interrupt



100    DmaRegs.CH1.MODE.bit.DATASIZE = datasize;       // 16-bit/32-bit data size transfers



101    DmaRegs.CH1.MODE.bit.CHINTMODE = chintmode;     // Generate interrupt to CPU at beginning/end of transfer



102    DmaRegs.CH1.MODE.bit.CHINTE = chinte;           // Channel Interrupt to CPU enable



103



104    // Clear any spurious flags:



105    DmaRegs.CH1.CONTROL.bit.PERINTCLR = 1;          // Clear any spurious interrupt flags



106    DmaRegs.CH1.CONTROL.bit.ERRCLR = 1;             // Clear any spurious sync error flags



107



108    // Initialize PIE vector for CPU interrupt:



109    PieCtrlRegs.PIEIER7.bit.INTx1 = 1;              // Enable DMA CH1 interrupt in PIE



110



111    EDIS;



112}

545void DMACH6ModeConfig(Uint16 persel, Uint16 perinte, Uint16 oneshot, Uint16 cont, Uint16 synce, Uint16 syncsel, Uint16 ovrinte, Uint16 datasize, Uint16 chintmode, Uint16 chinte)



546{



547    EALLOW;



548



549        // Set up MODE Register:



550    DmaClaSrcSelRegs.DMACHSRCSEL2.bit.CH6 = persel; // Source select



551    DmaRegs.CH6.MODE.bit.PERINTSEL = 6;             // Should be hard coded to channel, above now selects source



552    DmaRegs.CH6.MODE.bit.PERINTE = perinte;         // Peripheral interrupt enable



553    DmaRegs.CH6.MODE.bit.ONESHOT = oneshot;         // Oneshot enable



554    DmaRegs.CH6.MODE.bit.CONTINUOUS = cont;         // Continuous enable



555    DmaRegs.CH6.MODE.bit.OVRINTE = ovrinte;         // Enable/disable the overflow interrupt



556    DmaRegs.CH6.MODE.bit.DATASIZE = datasize;       // 16-bit/32-bit data size transfers



557    DmaRegs.CH6.MODE.bit.CHINTMODE = chintmode;     // Generate interrupt to CPU at beginning/end of transfer



558    DmaRegs.CH6.MODE.bit.CHINTE = chinte;           // Channel Interrupt to CPU enable



559



560    // Clear any spurious flags:



561    DmaRegs.CH6.CONTROL.bit.PERINTCLR = 1;          // Clear any spurious interrupt flags



562    DmaRegs.CH6.CONTROL.bit.ERRCLR = 1;             // Clear any spurious sync error flags



563



564    // Initialize PIE vector for CPU interrupt:



565    PieCtrlRegs.PIEIER7.bit.INTx6 = 1;              // Enable DMA CH6 interrupt in PIE



566



567    EDIS;



568}
