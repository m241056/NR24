54void initSetGPIOIPC(void)



55{



56



57    uint32_t *pulMsgRam;



58    volatile uint32_t ipcCountVar = 0;



59    #ifdef CPU1



60	uint32_t i = 0;



61	uint32_t MemCfgRegVar = 0;



62        EALLOW; // This is needed to write to EALLOW protected registers



63        PieVectTable.IPC0_INT = &CPU02toCPU01IPC0IntHandler;



64        /*Each global shared RAM section size is 4096 (2^12)*/



65        for (i=((uint32_t)CPU2_RAMGS_DATA_START) >> 12; i < (((uint32_t)CPU2_RAMGS_DATA_LENGTH + (uint32_t)CPU2_RAMGS_DATA_START) >> 12); i++)



66        {



67            /*Global shred RAM strts at 0xC000*/



68            MemCfgRegVar = MemCfgRegVar | (1UL<<(i-12));



69        }



70        for (i=((uint32_t)CPU2_RAMGS_PROG_START) >> 12; i < (((uint32_t)CPU2_RAMGS_PROG_LENGTH + (uint32_t)CPU2_RAMGS_PROG_START) >> 12); i++)



71        {



72            MemCfgRegVar = MemCfgRegVar | (1UL<<(i-12));



73        }



74        MemCfgRegs.GSxMSEL.all = MemCfgRegs.GSxMSEL.all | MemCfgRegVar;



75        EDIS;    // This is needed to disable write to EALLOW protected registers



76        pulMsgRam = (void *)CPU01_TO_CPU02_PASSMSG;



77        pulMsgRam[0] = (uint32_t)&setGPIOForCPU2;



78        pulMsgRam[1] = (uint32_t)&mwInitStart;



79        pulMsgRam[2] = (uint32_t)&mwInitEnd;



80        pulMsgRam[3] = (uint32_t)&setInXBARForCPU2;



81        pulMsgRam[4] = (uint32_t)&setOutXBARForCPU2;



82        pulMsgRam[5] = (uint32_t)&setEpwmXBARForCPU2;



83        PieCtrlRegs.PIEIER1.bit.INTx13 = 1;     // IPC INT0



84        IER |= M_INT1;



85        EINT;  // Enable Global interrupt INTM



86        ERTM;  // Enable Global realtime interrupt DBGM



87        



88    //#if MW_RUNTIME_FLASHLOAD



89      #ifndef TIC2000_SOC_BASEPRODUCT



90        //Wait for CPU2 to get ready to accept boot command from CPU1



91        while (((IpcRegs.IPCBOOTSTS & 0x0000000F) != C2_BOOTROM_BOOTSTS_SYSTEM_READY) && (700 > ipcCountVar))



92        {



93            ipcCountVar++;



94        }



95        //CPU01 to CPU02 IPC Boot Mode Register



96        IpcRegs.IPCBOOTMODE = C1C2_BROM_BOOTMODE_BOOT_FROM_FLASH;



97        // CPU01 To CPU02 IPC Command Register



98        IpcRegs.IPCSENDCOM  = BROM_IPC_EXECUTE_BOOTMODE_CMD;



99        // CPU01 to CPU02 IPC flag register



100        IpcRegs.IPCSET.all = 0x80000001;



101        ipcCountVar = 0;



102        // Give ~35 usec delay to start CPU2 init



103        while (500 > ipcCountVar) {



104            ipcCountVar++;



105        }



106      #endif



107



108    //#endif



109



110    #else



111        #ifndef TIC2000_SOC_BASEPRODUCT



112        pulMsgRam = (void *)CPU01_TO_CPU02_PASSMSG;



113        IPCLiteLtoRDataWrite(IPC_FLAG0, pulMsgRam[1], 0x425,



114                         IPC_LENGTH_32_BITS, IPC_FLAG31);



115        #endif



116    #endif



117}

120Uint32 setGPIOForCPU2(Uint32 gpioData)



121{



122    if (((gpioData & 0xFF000000)>>24) == GPIO_MUX_CPU2CLA) /* from CLA inside CPU2, GPIO_MUX_CPU2CLA defined to value 3*/



123    {



124        gpioData = gpioData & 0x00FFFFFF; /* remove added encryption GPIO_MUX_CPU2CLA, MSB is the flag and we use only flag 0*/



125        GPIO_SetupPinMux(gpioData & 0x000000FF, GPIO_MUX_CPU2CLA, (gpioData & 0x0000FF00)>>8);



126        GPIO_SetupPinOptions(gpioData & 0x000000FF, (gpioData & 0x00FF0000)>>16, (gpioData & 0xFF000000)>>24);



127    }



128    else



129    {



130        GPIO_SetupPinMux(gpioData & 0x000000FF, GPIO_MUX_CPU2, (gpioData & 0x0000FF00)>>8);/* From CPU2*/



131        GPIO_SetupPinOptions(gpioData & 0x000000FF, (gpioData & 0x00FF0000)>>16, (gpioData & 0xFF000000)>>24);



132    }



133    return 1;



134}

189Uint32 setOutXBARForCPU2(Uint32 inData)



190{



191    volatile Uint32 *muxEnableRegAddr;



192    volatile Uint32 *muxRegAddr;



193    volatile Uint32 *muxLatchRegAddr;



194    volatile Uint32 *muxInvertRegAddr;



195    Uint16 latch = (Uint16)((inData & 0xF0000000) >> 28);



196    Uint16 invert = (Uint16)((inData & 0x0F000000) >> 24);



197    Uint16 muxVal = (Uint16)((inData & 0x00FF0000) >> 16);



198    Uint16 mux = (Uint16)((inData & 0x0000FF00) >> 8);



199    Uint16 xbar = (Uint16)(inData & 0x000000FF);



200    Uint16 muxPosition = mux%16;



201    



202    muxRegAddr          = (Uint32 *)&OutputXbarRegs + ((2*xbar)-2) + (mux/16);



203    muxEnableRegAddr    = (Uint32 *)&OutputXbarRegs + 0x10 + (xbar-1);



204    muxLatchRegAddr    = (Uint32 *)&OutputXbarRegs + 0x1B;



205    muxInvertRegAddr    = (Uint32 *)&OutputXbarRegs + 0x1C;



206    EALLOW;



207    *muxRegAddr        |= (Uint32)(muxVal & 0x3) << (2*muxPosition);



208    *muxEnableRegAddr  |= (Uint32)1 << mux;



209    *muxLatchRegAddr   |= (Uint32)(latch & 0x1) << (xbar-1);



210    *muxInvertRegAddr  |= (Uint32)(invert & 0x1) << (xbar-1);



211    EDIS;



212    return 1;



213}

216{



217    volatile Uint32 *muxEnableRegAddr;



218    volatile Uint32 *muxRegAddr;



219    volatile Uint32 *muxInvertRegAddr;



220    Uint16 invert = (Uint16)((inData & 0x0F000000) >> 24);



221    Uint16 muxVal = (Uint16)((inData & 0x00FF0000) >> 16);



222    Uint16 mux = (Uint16)((inData & 0x0000FF00) >> 8);



223    Uint16 xbar = (Uint16)(inData & 0x000000FF);



224    Uint16 muxPosition = mux%16;



225    



226    muxRegAddr          = (Uint32 *)&EPwmXbarRegs + ((2*xbar)-2) + (mux/16);



227    muxEnableRegAddr    = (Uint32 *)&EPwmXbarRegs + 0x10 + (xbar-1);



228    muxInvertRegAddr    = (Uint32 *)&EPwmXbarRegs + 0x1C;



229    EALLOW;



230    *muxRegAddr        |= (Uint32)(muxVal & 0x3) << (2*muxPosition);



231    *muxEnableRegAddr  |= (Uint32)1 << mux;



232    *muxInvertRegAddr  |= (Uint32)(invert & 0x1) << (xbar-1);



233    EDIS;



234    return 1;



235}

237__interrupt void CPU02toCPU01IPC0IntHandler (void)



238{



239    // Continue processing messages



240    uint32_t command;



241    command = IpcRegs.IPCRECVCOM;



242    



243    switch (command)



244    {



245    case IPC_SET_BITS_16:



246        IPCLiteRtoLSetBits(IPC_FLAG0, IPC_FLAG31);



247        break;



248    case IPC_SET_BITS_32:



249        IPCLiteRtoLSetBits(IPC_FLAG0, IPC_FLAG31);



250        break;



251    case IPC_CLEAR_BITS_16:



252        IPCLiteRtoLClearBits(IPC_FLAG0, IPC_FLAG31);



253        break;



254    case IPC_CLEAR_BITS_32:



255        IPCLiteRtoLClearBits(IPC_FLAG0, IPC_FLAG31);



256        break;



257    case IPC_DATA_WRITE_16:



258        IPCLiteRtoLDataWrite(IPC_FLAG0, IPC_FLAG31);



259        break;



260    case IPC_DATA_WRITE_32:



261        IPCLiteRtoLDataWrite(IPC_FLAG0, IPC_FLAG31);



262        break;



263    case IPC_SET_BITS_16_PROTECTED:



264        IPCLiteRtoLSetBits_Protected(IPC_FLAG0, IPC_FLAG31);



265        break;



266    case IPC_SET_BITS_32_PROTECTED:



267        IPCLiteRtoLSetBits_Protected(IPC_FLAG0, IPC_FLAG31);



268        break;



269    case IPC_CLEAR_BITS_16_PROTECTED:



270        IPCLiteRtoLClearBits_Protected(IPC_FLAG0, IPC_FLAG31);



271        break;



272    case IPC_CLEAR_BITS_32_PROTECTED:



273        IPCLiteRtoLClearBits_Protected(IPC_FLAG0, IPC_FLAG31);



274        break;



275    case IPC_DATA_WRITE_16_PROTECTED:



276        IPCLiteRtoLDataWrite_Protected(IPC_FLAG0, IPC_FLAG31);



277        break;



278    case IPC_DATA_WRITE_32_PROTECTED:



279        IPCLiteRtoLDataWrite_Protected(IPC_FLAG0, IPC_FLAG31);



280        break;



281    case IPC_DATA_READ_16:



282        IPCLiteRtoLDataRead(IPC_FLAG0, IPC_FLAG31);



283        break;



284    case IPC_DATA_READ_32:



285        IPCLiteRtoLDataRead(IPC_FLAG0, IPC_FLAG31);



286        break;



287    case IPC_FUNC_CALL:



288        IPCLiteRtoLFunctionCall(IPC_FLAG0, IPC_FLAG31);



289        break;



290    default:



291        ErrorFlag = 1;



292        break;



293    }



294



295    PieCtrlRegs.PIEACK.all = PIEACK_GROUP1;   // Acknowledge interrupt to PIE



296    // try re-mapping the IPC INT0.



297    if (IpcRegs.IPCRECVDATA == 0x36E) // mwInitEnd



298    {



299        IpcRegs.IPCACK.all |= IPC_FLAG0;



300        IpcRegs.IPCACK.all |= IPC_FLAG31;



301        IpcRegs.IPCCLR.all |= IPC_FLAG0;



302        IpcRegs.IPCCLR.all |= IPC_FLAG31;



303        #ifdef REMAPIPCINT0



304           EALLOW;



305           PieVectTable.IPC0_INT = &IPC0_INT_isr;/* Hook interrupt to the ISR*/



306           EDIS;



307        #else



308           asm(" SETC INTM");



309           PieCtrlRegs.PIEIER1.bit.INTx13 = 0;     // IPC INT0 disable



310           asm(" RPT #5 || NOP");



311           IFR &= 0xFFFE;



312           PieCtrlRegs.PIEACK.all = 0x1;



313           asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



314        #endif



315    }



316}
