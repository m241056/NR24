131            }



132            case 1:



133            {



134                lIntNumber = INT_CANA_1;



135                break;



136



137            }



138            default:



139            {



140                lIntNumber = -1;



141                break;



142            }



143            }



144            break;



145        }



146



147        // Return the interrupt number for CAN 1



148        case CANB_BASE:



149        {



150            switch(ucNumber)



151            {



152            case 0:



153            {



154                lIntNumber = INT_CANB_0;



155                break;



156            }



157            case 1:



158            {



159                lIntNumber = INT_CANB_1;



160                break;



161



162            }



163            default:



164            {



165                lIntNumber = -1;



166                break;



167            }



168            }



169            break;



170        }



171



172        // Return -1 to indicate a bad address was passed in.



173        default:



174        {



175            lIntNumber = -1;



176        }



177    }



178    return(lIntNumber);



179}

181//*****************************************************************************



182//! \internal



183//!



184//! Copies data from a buffer to the CAN Data registers.



185//!



186//! \param pucData is a pointer to the data to be written out to the CAN



187//! controller's data registers.



188//! \param pui32Register is an uint32_t pointer to the first register of the



189//! CAN controller's data registers.  For example, in order to use the IF1



190//! register set on CAN controller A, the value would be: \b CANA_BASE \b +



191//! \b CAN_O_IF1DATA.



192//! \param iSize is the number of bytes to copy into the CAN controller.



193//!



194//! This function takes the steps necessary to copy data from a contiguous



195//! buffer in memory into the non-contiguous data registers used by the CAN



196//! controller.  This function is rarely used outside of the CANMessageSet()



197//! function.



198//!



199//! This function replaces the original CANWriteDataReg() API and performs the



200//! same actions.  A macro is provided in <tt>can.h</tt> to map the original



201//! API to this API.



202//!



203//! \return None.



204//



205//*****************************************************************************



206static void



207CANDataRegWrite(unsigned char *pucData, uint32_t *pui32Register, int16_t iSize)



208{



209    int16_t iIdx;



210    unsigned char * pucRegister = (unsigned char *) pui32Register;



211



212    // Loop always copies 1 or 2 bytes per iteration.



213    for(iIdx = 0; iIdx < iSize; iIdx++ )



214    {



215        // Write out the data 8 bits at a time.



216        HWREGB(pucRegister++) = pucData[iIdx];



217    }



218}

222//!

224//!



225//! \param pucData is a pointer to the location to store the data read from the



226//! CAN controller's data registers.



227//! \param pui32Register is an uint32_t pointer to the first register of the



228//! CAN controller's data registers.  For example, in order to use the IF1



229//! register set on CAN controller A, the value would be: \b CANA_BASE \b +



230//! \b CAN_O_IF1DATA.



231//! \param iSize is the number of bytes to copy from the CAN controller.



232//!



233//! This function takes the steps necessary to copy data to a contiguous buffer



234//! in memory from the non-contiguous data registers used by the CAN



235//! controller.  This function is rarely used outside of the CANMessageGet()

237//!



238//! This function replaces the original CANReadDataReg() API and performs the

242//! \return None.

244//*****************************************************************************



245static void



246CANDataRegRead(unsigned char *pucData, uint32_t *pui32Register, int16_t iSize)



247{



248    int16_t iIdx;



249    unsigned char * pucRegister = (unsigned char *) pui32Register;



250



251    // Loop always copies 1 or 2 bytes per iteration.



252    for(iIdx = 0; iIdx < iSize; iIdx++ )



253    {



254        // Read out the data



255        pucData[iIdx] = HWREGB(pucRegister++);



256    }



257}

259//*****************************************************************************



260//



261//! Initializes the CAN controller after reset.



262//!



263//! \param ui32Base is the base address of the CAN controller.



264//!



265//! After reset, the CAN controller is left in the disabled state.  However,



266//! the memory used for message objects contains undefined values and must be



267//! cleared prior to enabling the CAN controller the first time.  This prevents



268//! unwanted transmission or reception of data before the message objects are



269//! configured.  This function must be called before enabling the controller



270//! the first time.



271//!



272//! \return None.



273//



274//*****************************************************************************



275void



276CANInit(uint32_t ui32Base)



277{



278    int16_t iMsg;



279



280    // Check the arguments.



281    ASSERT(CANBaseValid(ui32Base));



282



283    // Place CAN controller in init state, regardless of previous state.  This



284    // will put controller in idle, and allow the message object RAM to be



285    // programmed.



286



287    HWREGH(ui32Base + CAN_O_CTL) = CAN_CTL_INIT;



288    HWREGH(ui32Base + CAN_O_CTL) = CAN_CTL_SWR;



289



290    // Wait for busy bit to clear



291    while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



292    {



293    }



294



295    // Clear the message value bit in the arbitration register.  This indicates



296    // the message is not valid and is a "safe" condition to leave the message



297    // object.  The same arb reg is used to program all the message objects.



298    HWREGH(ui32Base + CAN_O_IF1CMD + 2) = (CAN_IF1CMD_DIR | CAN_IF1CMD_ARB |



299                                           CAN_IF1CMD_CONTROL) >> 16;



300    HWREGH(ui32Base + CAN_O_IF1ARB) = 0;



301    HWREGH(ui32Base + CAN_O_IF1ARB + 2) = 0;



302



303    HWREGH(ui32Base + CAN_O_IF1MCTL) = 0;



304    HWREGH(ui32Base + CAN_O_IF1MCTL + 2) = 0;



305



306    HWREGH(ui32Base + CAN_O_IF2CMD + 2) = (CAN_IF2CMD_DIR | CAN_IF2CMD_ARB |



307                                           CAN_IF2CMD_CONTROL) >> 16;



308    HWREGH(ui32Base + CAN_O_IF2ARB) = 0;



309    HWREGH(ui32Base + CAN_O_IF2ARB + 2) = 0;



310



311    HWREGH(ui32Base + CAN_O_IF2MCTL) = 0;



312    HWREGH(ui32Base + CAN_O_IF2MCTL + 2) = 0;



313



314    // Loop through to program all 32 message objects



315    for(iMsg = 1; iMsg <= 32; iMsg+=2)



316    {



317        // Wait for busy bit to clear



318        while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



319        {



320        }



321



322        // Initiate programming the message object



323        HWREGH(ui32Base + CAN_O_IF1CMD) = iMsg;



324



325        // Wait for busy bit to clear



326        while(HWREGH(ui32Base + CAN_O_IF2CMD) & CAN_IF2CMD_BUSY)



327        {



328        }



329



330        // Initiate programming the message object



331        HWREGH(ui32Base + CAN_O_IF2CMD) = iMsg + 1;



332    }



333



334    // Make sure that the interrupt and new data flags are updated for the



335    // message objects.



336    HWREGH(ui32Base + CAN_O_IF1CMD + 2) = (CAN_IF1CMD_TXRQST |



337                                           CAN_IF1CMD_CLRINTPND) >> 16;



338    HWREGH(ui32Base + CAN_O_IF2CMD + 2) = (CAN_IF2CMD_TXRQST |



339                                           CAN_IF2CMD_CLRINTPND) >> 16;



340



341    // Loop through to program all 32 message objects



342    for(iMsg = 1; iMsg <= 32; iMsg+=2)



343    {



344        // Wait for busy bit to clear.



345        while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



346        {



347        }



348



349        // Initiate programming the message object



350        HWREGH(ui32Base + CAN_O_IF1CMD) = iMsg;



351



352        // Wait for busy bit to clear.



353        while(HWREGH(ui32Base + CAN_O_IF2CMD) & CAN_IF2CMD_BUSY)



354        {



355        }



356



357        // Initiate programming the message object



358        HWREGH(ui32Base + CAN_O_IF2CMD) = iMsg + 1;



359    }



360



361    // Acknowledge any pending status interrupts.



362    HWREG(ui32Base + CAN_O_ES);



363}

456//*****************************************************************************



457//



458//! Reads the current settings for the CAN controller bit timing.



459//!



460//! \param ui32Base is the base address of the CAN controller.



461//! \param pClkParms is a pointer to a structure to hold the timing parameters.



462//!



463//! This function reads the current configuration of the CAN controller bit



464//! clock timing, and stores the resulting information in the structure



465//! supplied by the caller.  Refer to CANBitTimingSet() for the meaning of the



466//! values that are returned in the structure pointed to by \e pClkParms.



467//!



468//! This function replaces the original CANGetBitTiming() API and performs the



469//! same actions.  A macro is provided in <tt>can.h</tt> to map the original



470//! API to this API.



471//!



472//! \return None.



473//



474//*****************************************************************************



475void



476CANBitTimingGet(uint32_t ui32Base, tCANBitClkParms *pClkParms)



477{



478    uint32_t uBitReg;



479



480    // Check the arguments.



481    ASSERT(CANBaseValid(ui32Base));



482    ASSERT(pClkParms != 0);



483



484    uBitReg = HWREG(ui32Base + CAN_O_BTR);



485



486    // Set the phase 2 segment.



487    pClkParms->uPhase2Seg = ((uBitReg & CAN_BTR_TSEG2_M) >> 12) + 1;



488



489    // Set the phase 1 segment.



490    pClkParms->uSyncPropPhase1Seg = ((uBitReg & CAN_BTR_TSEG1_M) >> 8) + 1;



491



492    // Set the synchronous jump width.



493    pClkParms->uSJW = ((uBitReg & CAN_BTR_SJW_M) >> 6) + 1;



494



495    // Set the pre-divider for the CAN bus bit clock.



496    pClkParms->uQuantumPrescaler = ((uBitReg & CAN_BTR_BRP_M) |



497                                    ((uBitReg & CAN_BTR_BRPE_M) >> 10)) + 1;



498}

500//*****************************************************************************



501//



502//! This function is used to set the CAN bit timing values to a nominal setting



503//! based on a desired bit rate.



504//!



505//! \param ui32Base is the base address of the CAN controller.



506//! \param ui32SourceClock is the clock frequency for the CAN peripheral in Hz.



507//! \param ui32BitRate is the desired bit rate.



508//!



509//! This function will set the CAN bit timing for the bit rate passed in the



510//! \e ui32BitRate parameter based on the \e ui32SourceClock parameter. The CAN



511//! bit clock is calculated to be an average timing value that should work for



512//! most systems.  If tighter timing requirements are needed, then the



513//! CANBitTimingSet() function is available for full customization of all of



514//! the CAN bit timing values.  Since not all bit rates can be matched



515//! exactly, the bit rate is set to the value closest to the desired bit rate



516//! without being higher than the \e ui32BitRate value.



517//!



518//! \return This function returns the bit rate that the CAN controller was



519//! configured to use or it returns 0 to indicate that the bit rate was not



520//! changed because the requested bit rate was not valid.



521//



522//*****************************************************************************



523uint32_t



524CANBitRateSet(uint32_t ui32Base, uint32_t ui32SourceClock, uint32_t ui32BitRate)



525{



526    uint32_t ui32DesiredRatio;



527    uint32_t ui32CANBits;



528    uint32_t ui32PreDivide;



529    uint32_t ui32RegValue;



530    uint16_t ui16CANCTL;



531



532    ASSERT(ui32BitRate != 0);



533



534    // Calculate the desired clock rate.



535    ui32DesiredRatio = ui32SourceClock / ui32BitRate;



536



537    // If the ratio of CAN bit rate to processor clock is too small or too



538    // large then return 0 indicating that no bit rate was set.



539    ASSERT(ui32DesiredRatio <= (CAN_MAX_PRE_DIVISOR * CAN_MAX_BIT_DIVISOR));



540    ASSERT(ui32DesiredRatio >= (CAN_MIN_PRE_DIVISOR * CAN_MIN_BIT_DIVISOR));



541



542    // Make sure that the Desired Ratio is not too large.  This enforces the



543    // requirement that the bit rate is larger than requested.



544    if((ui32SourceClock / ui32DesiredRatio) > ui32BitRate)



545    {



546        ui32DesiredRatio += 1;



547    }



548



549    // Check all possible values to find a matching value.



550    while(ui32DesiredRatio <= CAN_MAX_PRE_DIVISOR * CAN_MAX_BIT_DIVISOR)



551    {



552        // Loop through all possible CAN bit divisors.



553        for(ui32CANBits = CAN_MAX_BIT_DIVISOR;



554            ui32CANBits >= CAN_MIN_BIT_DIVISOR;



555            ui32CANBits--)



556        {



557            // For a given CAN bit divisor save the pre divisor.



558            ui32PreDivide = ui32DesiredRatio / ui32CANBits;



559



560            // If the calculated divisors match the desired clock ratio then



561            // return these bit rate and set the CAN bit timing.



562            if((ui32PreDivide * ui32CANBits) == ui32DesiredRatio)



563            {



564                // Start building the bit timing value by adding the bit timing



565                // in time quanta.



566                ui32RegValue =



567                    g_ui16CANBitValues[ui32CANBits - CAN_MIN_BIT_DIVISOR];



568



569                // To set the bit timing register, the controller must be



570                // placed



571                // in init mode (if not already), and also configuration change



572                // bit enabled.  The state of the register should be saved



573                // so it can be restored.



574



575                ui16CANCTL = HWREGH(ui32Base + CAN_O_CTL);



576                HWREGH(ui32Base + CAN_O_CTL) = ui16CANCTL | CAN_CTL_INIT |



577                                               CAN_CTL_CCE;



578



579                // Now add in the pre-scalar on the bit rate.



580                ui32RegValue |= ((ui32PreDivide - 1) & CAN_BTR_BRP_M) |



581                				(((ui32PreDivide - 1) << 10) & CAN_BTR_BRPE_M);



582



583                // Set the clock bits in the and the bits of the



584                // pre-scalar.



585                HWREGH(ui32Base + CAN_O_BTR) = (ui32RegValue &



586                                                CAN_REG_WORD_MASK);



587                HWREGH(ui32Base + CAN_O_BTR + 2) = (ui32RegValue >> 16);



588



589                // Restore the saved CAN Control register.



590                HWREGH(ui32Base + CAN_O_CTL) = ui16CANCTL;



591



592                // Return the computed bit rate.



593                return(ui32SourceClock / ( ui32PreDivide * ui32CANBits));



594            }



595        }



596



597        // Move the divisor up one and look again.  Only in rare cases are



598        // more than 2 loops required to find the value.



599        ui32DesiredRatio++;



600    }



601    return(0);



602}

604//*****************************************************************************



605//



606//! Configures the CAN controller bit timing.



607//!



608//! \param ui32Base is the base address of the CAN controller.



609//! \param pClkParms points to the structure with the clock parameters.



610//!



611//! Configures the various timing parameters for the CAN bus bit timing:



612//! Propagation segment, Phase Buffer 1 segment, Phase Buffer 2 segment, and



613//! the Synchronization Jump Width.  The values for Propagation and Phase



614//! Buffer 1 segments are derived from the combination



615//! \e pClkParms->uSyncPropPhase1Seg parameter.  Phase Buffer 2 is determined



616//! from the \e pClkParms->uPhase2Seg parameter.  These two parameters, along



617//! with \e pClkParms->uSJW are based in units of bit time quanta.  The actual



618//! quantum time is determined by the \e pClkParms->uQuantumPrescaler value,



619//! which specifies the divisor for the CAN module clock.



620//!



621//! The total bit time, in quanta, will be the sum of the two Seg parameters,



622//! as follows:



623//!



624//! bit_time_q = uSyncPropPhase1Seg + uPhase2Seg + 1



625//!



626//! Note that the Sync_Seg is always one quantum in duration, and will be added



627//! to derive the correct duration of Prop_Seg and Phase1_Seg.



628//!



629//! The equation to determine the actual bit rate is as follows:



630//!



631//! CAN Clock /



632//! ((\e uSyncPropPhase1Seg + \e uPhase2Seg + 1) * (\e uQuantumPrescaler))



633//!



634//! This means that with \e uSyncPropPhase1Seg = 4, \e uPhase2Seg = 1,



635//! \e uQuantumPrescaler = 2 and an 8 MHz CAN clock, that the bit rate will be



636//! (8 MHz) / ((5 + 2 + 1) * 2) or 500 Kbit/sec.



637//!



638//! \return None.



639//



640//*****************************************************************************



641void



642CANBitTimingSet(uint32_t ui32Base, tCANBitClkParms *pClkParms)



643{



644    uint32_t uBitReg;



645    uint16_t uSavedInit;



646



647    // Check the arguments.



648    ASSERT(CANBaseValid(ui32Base));



649    ASSERT(pClkParms != 0);



650



651    // The phase 1 segment must be in the range from 2 to 16.



652    ASSERT((pClkParms->uSyncPropPhase1Seg >= 2) &&



653           (pClkParms->uSyncPropPhase1Seg <= 16));



654



655    // The phase 2 segment must be in the range from 1 to 8.



656    ASSERT((pClkParms->uPhase2Seg >= 1) && (pClkParms->uPhase2Seg <= 8));



657



658    // The synchronous jump windows must be in the range from 1 to 4.



659    ASSERT((pClkParms->uSJW >= 1) && (pClkParms->uSJW <= 4));



660



661    // The CAN clock pre-divider must be in the range from 1 to 1024.



662    ASSERT((pClkParms->uQuantumPrescaler <= 1024) &&



663           (pClkParms->uQuantumPrescaler >= 1));



664



665    // To set the bit timing register, the controller must be placed in init



666    // mode (if not already), and also configuration change bit enabled.  State



667    // of the init bit should be saved so it can be restored at the end.



668    uSavedInit = HWREGH(ui32Base + CAN_O_CTL);



669    HWREGH(ui32Base + CAN_O_CTL) = uSavedInit | CAN_CTL_INIT | CAN_CTL_CCE;



670



671    // Set the bit fields of the bit timing register according to the parms.



672    uBitReg = ((pClkParms->uPhase2Seg - 1) << 12) & CAN_BTR_TSEG2_M;



673    uBitReg |= ((pClkParms->uSyncPropPhase1Seg - 1) << 8) & CAN_BTR_TSEG1_M;



674    uBitReg |= ((pClkParms->uSJW - 1) << 6) & CAN_BTR_SJW_M;



675    uBitReg |= (pClkParms->uQuantumPrescaler - 1) & CAN_BTR_BRP_M;



676    uBitReg |= ((pClkParms->uQuantumPrescaler - 1) << 10)& CAN_BTR_BRPE_M;



677    HWREGH(ui32Base + CAN_O_BTR) = uBitReg & CAN_REG_WORD_MASK;



678    HWREGH(ui32Base + CAN_O_BTR + 2) = uBitReg >> 16;



679



680    // Clear the config change bit, and restore the init bit.



681    uSavedInit &= ~CAN_CTL_CCE;



682



683    // If Init was not set before, then clear it.



684    if(uSavedInit & CAN_CTL_INIT)



685    {



686        uSavedInit &= ~CAN_CTL_INIT;



687    }



688    HWREGH(ui32Base + CAN_O_CTL) = uSavedInit;



689}

854//!

861//!

878//!

894



895    // See which status the caller is looking for.

897    {

899        // specified by ui32Base.

901        {



902            ui32Status = HWREG(ui32Base + CAN_O_INT);



903            break;



904        }



905



906        // The caller wants the current message status interrupt for all



907        // messages.



908        case CAN_INT_STS_OBJECT:



909        {



910            // Read message object interrupt status



911            ui32Status = HWREG(ui32Base + CAN_O_IPEN_21);



912            break;



913        }



914



915        // Request was for unknown status so just return 0.



916        default:



917        {



918            ui32Status = 0;



919            break;



920        }



921    }



922    // Return the interrupt status value



923    return(ui32Status);

926//*****************************************************************************



927//



928//! Clears a CAN interrupt source.



929//!



930//! \param ui32Base is the base address of the CAN controller.



931//! \param ui32IntClr is a value indicating which interrupt source to clear.



932//!



933//! This function can be used to clear a specific interrupt source.  The



934//! \e ui32IntClr parameter should be one of the following values:



935//!



936//! - \b CAN_INT_INTID_STATUS - Clears a status interrupt.



937//! - 1-32 - Clears the specified message object interrupt



938//!



939//! It is not necessary to use this function to clear an interrupt.  This



940//! should only be used if the application wants to clear an interrupt source



941//! without taking the normal interrupt action.



942//!



943//! Normally, the status interrupt is cleared by reading the controller status



944//! using CANStatusGet().  A specific message object interrupt is normally



945//! cleared by reading the message object using CANMessageGet().



946//!



947//! \note Since there is a write buffer in the Cortex-M3 processor, it may take



948//! several clock cycles before the interrupt source is actually cleared.



949//! Therefore, it is recommended that the interrupt source be cleared early in



950//! the interrupt handler (as opposed to the very last action) to avoid



951//! returning from the interrupt handler before the interrupt source is



952//! actually cleared.  Failure to do so may result in the interrupt handler



953//! being immediately reentered (since NVIC still sees the interrupt source



954//! asserted).



955//!



956//! \return None.



957//



958//*****************************************************************************



959void



960CANIntClear(uint32_t ui32Base, uint32_t ui32IntClr)



961{



962    // Check the arguments.



963    ASSERT(CANBaseValid(ui32Base));



964    ASSERT((ui32IntClr == CAN_INT_INT0ID_STATUS) ||



965           ((ui32IntClr>=1) && (ui32IntClr <=32)));



966



967    if(ui32IntClr == CAN_INT_INT0ID_STATUS)



968    {



969        // Simply read and discard the status to clear the interrupt.



970        HWREG(ui32Base + CAN_O_ES);



971    }



972    else



973    {



974        // Wait to be sure that this interface is not busy.



975        while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



976        {



977        }



978



979        // Only change the interrupt pending state by setting only the



980        // CAN_IF1CMD_CLRINTPND bit.



981        HWREGH(ui32Base + CAN_O_IF1CMD + 2) = CAN_IF1CMD_CLRINTPND >> 16;



982



983        // Send the clear pending interrupt command to the CAN controller.



984        HWREGH(ui32Base + CAN_O_IF1CMD) = ui32IntClr & CAN_IF1CMD_MSG_NUM_M;



985



986        // Wait to be sure that this interface is not busy.



987        while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



988        {



989        }



990    }



991}

1295//!



1296//! \return None.



1297//



1298//*****************************************************************************



1299void



1300CANMessageSet(uint32_t ui32Base, uint32_t ui32ObjID, tCANMsgObject *pMsgObject,



1301              tMsgObjType eMsgType)



1302{



1303    uint32_t ui32CmdMaskReg;



1304    uint32_t ui32MaskReg;



1305    uint32_t ui32ArbReg;



1306    uint32_t ui32MsgCtrl;



1307    bool bTransferData;



1308    bool bUseExtendedID;



1309



1310    bTransferData = 0;



1311



1312    // Check the arguments.



1313    ASSERT(CANBaseValid(ui32Base));



1314    ASSERT((ui32ObjID <= 32) && (ui32ObjID != 0));



1315    ASSERT((eMsgType == MSG_OBJ_TYPE_TX) ||



1316           (eMsgType == MSG_OBJ_TYPE_TX_REMOTE) ||



1317           (eMsgType == MSG_OBJ_TYPE_RX) ||



1318           (eMsgType == MSG_OBJ_TYPE_RX_REMOTE) ||



1319           (eMsgType == MSG_OBJ_TYPE_TX_REMOTE) ||



1320           (eMsgType == MSG_OBJ_TYPE_RXTX_REMOTE));



1321



1322    // Wait for busy bit to clear



1323    while(HWREGH(ui32Base + CAN_O_IF1CMD) & CAN_IF1CMD_BUSY)



1324    {



1325    }



1326



1327    // See if we need to use an extended identifier or not.



1328    if((pMsgObject->ui32MsgID > CAN_MAX_11BIT_MSG_ID) ||



1329       (pMsgObject->ui32Flags & MSG_OBJ_EXTENDED_ID))



1330    {



1331        bUseExtendedID = 1;



1332    }



1333    else



1334    {



1335        bUseExtendedID = 0;



1336    }



1337



1338    // This is always a write to the Message object as this call is setting a



1339    // message object.  This call will also always set all size bits so it sets



1340    // both data bits.  The call will use the CONTROL register to set control



1341    // bits so this bit needs to be set as well.



1342    ui32CmdMaskReg = (CAN_IF1CMD_DIR | CAN_IF1CMD_DATA_A | CAN_IF1CMD_DATA_B |



1343                      CAN_IF1CMD_CONTROL);



1344



1345    // Initialize the values to a known state before filling them in based on



1346    // the type of message object that is being configured.



1347    ui32ArbReg = 0;



1348    ui32MsgCtrl = 0;



1349    ui32MaskReg = 0;



1350



1351    switch(eMsgType)



1352    {



1353    // Transmit message object.



1354    case MSG_OBJ_TYPE_TX:



1355    {



1356        // Set the TXRQST bit and the reset the rest of the register.



1357        ui32MsgCtrl |= CAN_IF1MCTL_TXRQST;



1358        ui32ArbReg = CAN_IF1ARB_DIR;



1359        bTransferData = 1;



1360        break;



1361    }



1362



1363    // Transmit remote request message object



1364    case MSG_OBJ_TYPE_TX_REMOTE:



1365    {



1366        // Set the TXRQST bit and the reset the rest of the register.



1367        ui32MsgCtrl |= CAN_IF1MCTL_TXRQST;



1368        ui32ArbReg = 0;



1369        break;



1370    }



1371



1372    // Receive message object.



1373    case MSG_OBJ_TYPE_RX:



1374    {



1375        // This clears the DIR bit along with everything else.  The TXRQST



1376        // bit was cleared by defaulting ui32MsgCtrl to 0.



1377        ui32ArbReg = 0;



1378        break;



1379    }



1380



1381    // Receive remote request message object.



1382    case MSG_OBJ_TYPE_RX_REMOTE:



1383    {



1384        // The DIR bit is set to one for remote receivers.  The TXRQST bit



1385        // was cleared by defaulting ui32MsgCtrl to 0.



1386        ui32ArbReg = CAN_IF1ARB_DIR;



1387



1388        // Set this object so that it only indicates that a remote frame



1389        // was received and allow for software to handle it by sending back



1390        // a data frame.



1391        ui32MsgCtrl = CAN_IF1MCTL_UMASK;



1392



1393        // Use the full Identifier by default.



1394        ui32MaskReg = CAN_IF1MSK_MSK_M;



1395



1396        // Make sure to send the mask to the message object.



1397        ui32CmdMaskReg |= CAN_IF1CMD_MASK;



1398        break;



1399    }



1400



1401    // Remote frame receive remote, with auto-transmit message object.



1402    case MSG_OBJ_TYPE_RXTX_REMOTE:



1403    {



1404        // Oddly the DIR bit is set to one for remote receivers.



1405        ui32ArbReg = CAN_IF1ARB_DIR;



1406



1407        // Set this object to auto answer if a matching identifier is seen.



1408        ui32MsgCtrl = CAN_IF1MCTL_RMTEN | CAN_IF1MCTL_UMASK;



1409



1410        // The data to be returned needs to be filled in.



1411        bTransferData = 1;



1412        break;



1413    }



1414



1415    // This case should never happen due to the ASSERT statement at the



1416    // beginning of this function.



1417    default:



1418    {



1419        return;



1420    }



1421    }



1422



1423    // Configure the Mask Registers.



1424    if(pMsgObject->ui32Flags & MSG_OBJ_USE_ID_FILTER)



1425    {



1426        if(bUseExtendedID)



1427        {



1428            // Set the 29 bits of Identifier mask that were requested.



1429            ui32MaskReg = pMsgObject->ui32MsgIDMask & CAN_IF1MSK_MSK_M;



1430        }



1431        else



1432        {



1433



1434            // Put the 11 bit Mask Identifier into the upper bits of the field



1435            // in the register.



1436            ui32MaskReg = ((pMsgObject->ui32MsgIDMask << CAN_IF1ARB_STD_ID_S) &



1437                           CAN_IF1ARB_STD_ID_M);



1438        }



1439    }



1440



1441    // If the caller wants to filter on the extended ID bit then set it.



1442    if((pMsgObject->ui32Flags & MSG_OBJ_USE_EXT_FILTER) ==



1443       MSG_OBJ_USE_EXT_FILTER)



1444    {



1445        ui32MaskReg |= CAN_IF1MSK_MXTD;



1446    }



1447



1448    // The caller wants to filter on the message direction field.



1449    if((pMsgObject->ui32Flags & MSG_OBJ_USE_DIR_FILTER) ==



1450       MSG_OBJ_USE_DIR_FILTER)



1451    {



1452        ui32MaskReg |= CAN_IF1MSK_MDIR;



1453    }



1454



1455    if(pMsgObject->ui32Flags & (MSG_OBJ_USE_ID_FILTER | MSG_OBJ_USE_DIR_FILTER |



1456                              MSG_OBJ_USE_EXT_FILTER))



1457    {



1458        // Set the UMASK bit to enable using the mask register.



1459        ui32MsgCtrl |= CAN_IF1MCTL_UMASK;



1460



1461        // Set the MASK bit so that this gets transferred to the Message



1462        // Object.



1463        ui32CmdMaskReg |= CAN_IF1CMD_MASK;



1464    }



1465



1466    // Set the Arb bit so that this gets transferred to the Message object.



1467    ui32CmdMaskReg |= CAN_IF1CMD_ARB;



1468



1469    // Configure the Arbitration registers.



1470    if(bUseExtendedID)



1471    {



1472        // Set the 29 bit version of the Identifier for this message object.



1473        // Mark the message as valid and set the extended ID bit.



1474        ui32ArbReg |= (pMsgObject->ui32MsgID & CAN_IF1ARB_ID_M) |



1475                      CAN_IF1ARB_MSGVAL | CAN_IF1ARB_XTD;



1476    }



1477    else



1478    {



1479        // Set the 11 bit version of the Identifier for this message object.



1480        // The lower 18 bits are set to zero.



1481        // Mark the message as valid.



1482        ui32ArbReg |= ((pMsgObject->ui32MsgID << CAN_IF1ARB_STD_ID_S) &



1483                       CAN_IF1ARB_STD_ID_M) | CAN_IF1ARB_MSGVAL;



1484    }



1485



1486    // Set the data length since this is set for all transfers.  This is also a



1487    // single transfer and not a FIFO transfer so set EOB bit.



1488    ui32MsgCtrl |= (pMsgObject->ui32MsgLen & CAN_IF1MCTL_DLC_M);



1489



1490    // Mark this as the last entry if this is not the last entry in a FIFO.



1491    if((pMsgObject->ui32Flags & MSG_OBJ_FIFO) == 0)



1492    {



1493        ui32MsgCtrl |= CAN_IF1MCTL_EOB;



1494    }



1495



1496    // Enable transmit interrupts if they should be enabled.



1497    if(pMsgObject->ui32Flags & MSG_OBJ_TX_INT_ENABLE)



1498    {



1499        ui32MsgCtrl |= CAN_IF1MCTL_TXIE;



1500    }



1501



1502    // Enable receive interrupts if they should be enabled.



1503    if(pMsgObject->ui32Flags & MSG_OBJ_RX_INT_ENABLE)



1504    {



1505        ui32MsgCtrl |= CAN_IF1MCTL_RXIE;



1506    }



1507



1508    // Write the data out to the CAN Data registers if needed.



1509    if(bTransferData)



1510    {



1511        CANDataRegWrite(pMsgObject->pucMsgData,



1512                        (uint32_t *)(ui32Base + CAN_O_IF1DATA),



1513                        pMsgObject->ui32MsgLen);



1514    }



1515



1516    // Write out the registers to program the message object.



1517    HWREGH(ui32Base + CAN_O_IF1CMD + 2) = ui32CmdMaskReg >> 16;



1518



1519    HWREGH(ui32Base + CAN_O_IF1MSK) = ui32MaskReg & CAN_REG_WORD_MASK;



1520    HWREGH(ui32Base + CAN_O_IF1MSK + 2) = ui32MaskReg >> 16;



1521



1522    HWREGH(ui32Base + CAN_O_IF1ARB) = ui32ArbReg & CAN_REG_WORD_MASK;



1523    HWREGH(ui32Base + CAN_O_IF1ARB + 2) = ui32ArbReg >> 16;



1524



1525    HWREGH(ui32Base + CAN_O_IF1MCTL) = ui32MsgCtrl & CAN_REG_WORD_MASK;



1526



1527    // Transfer the message object to the message object specific by ui32ObjID.



1528    HWREGH(ui32Base + CAN_O_IF1CMD) = ui32ObjID & CAN_IF1CMD_MSG_NUM_M;



1529



1530    return;



1531}

1533//*****************************************************************************



1534//



1535//! Reads a CAN message from one of the message object buffers.



1536//!



1537//! \param ui32Base is the base address of the CAN controller.



1538//! \param ui32ObjID is the object number to read (1-32).



1539//! \param pMsgObject points to a structure containing message object fields.



1540//! \param bClrPendingInt indicates whether an associated interrupt should be



1541//! cleared.



1542//!



1543//! This function is used to read the contents of one of the 32 message objects



1544//! in the CAN controller, and return it to the caller.  The data returned is



1545//! stored in the fields of the caller-supplied structure pointed to by



1546//! \e pMsgObject.  The data consists of all of the parts of a CAN message,



1547//! plus some control and status information.



1548//!



1549//! Normally this is used to read a message object that has received and stored



1550//! a CAN message with a certain identifier.  However, this could also be used



1551//! to read the contents of a message object in order to load the fields of the



1552//! structure in case only part of the structure needs to be changed from a



1553//! previous setting.



1554//!



1555//! When using CANMessageGet, all of the same fields of the structure are



1556//! populated in the same way as when the CANMessageSet() function is used,



1557//! with the following exceptions:



1558//!



1559//! \e pMsgObject->ui32Flags:



1560//!



1561//! - \b MSG_OBJ_NEW_DATA indicates if this is new data since the last time it



1562//! was read



1563//! - \b MSG_OBJ_DATA_LOST indicates that at least one message was received on



1564//! this message object, and not read by the host before being overwritten.



1565//!



1566//! \return None.



1567//



1568//*****************************************************************************



1569void



1570CANMessageGet(uint32_t ui32Base, uint32_t ui32ObjID, tCANMsgObject *pMsgObject,



1571              bool bClrPendingInt)



1572{



1573    uint32_t ui32CmdMaskReg;



1574    uint32_t ui32MaskReg;



1575    uint32_t ui32ArbReg;



1576    uint32_t ui32MsgCtrl;



1577



1578    // Check the arguments.



1579    ASSERT(CANBaseValid(ui32Base));



1580    ASSERT((ui32ObjID <= 32) && (ui32ObjID != 0));



1581



1582    // This is always a read to the Message object as this call is setting a



1583    // message object.



1584    ui32CmdMaskReg = (CAN_IF2CMD_DATA_A | CAN_IF2CMD_DATA_B |



1585                      CAN_IF2CMD_CONTROL | CAN_IF2CMD_MASK | CAN_IF2CMD_ARB);



1586



1587    // Clear a pending interrupt and new data in a message object.



1588    if(bClrPendingInt)



1589    {



1590        ui32CmdMaskReg |= CAN_IF2CMD_CLRINTPND | CAN_IF2CMD_TXRQST;



1591    }



1592



1593    // Set up the request for data from the message object.



1594    HWREGH(ui32Base + CAN_O_IF2CMD + 2) =  ui32CmdMaskReg >> 16;



1595



1596    // Transfer the message object to the message object specified by ui32ObjID.



1597    HWREGH(ui32Base + CAN_O_IF2CMD) = ui32ObjID & CAN_IF2CMD_MSG_NUM_M;



1598



1599    // Wait for busy bit to clear



1600    while(HWREGH(ui32Base + CAN_O_IF2CMD) & CAN_IF2CMD_BUSY)



1601    {



1602    }



1603



1604    // Read out the IF Registers.



1605    ui32MaskReg = HWREG(ui32Base + CAN_O_IF2MSK);



1606    ui32ArbReg = HWREG(ui32Base + CAN_O_IF2ARB);



1607    ui32MsgCtrl = HWREG(ui32Base + CAN_O_IF2MCTL);



1608    pMsgObject->ui32Flags = MSG_OBJ_NO_FLAGS;



1609



1610    // Determine if this is a remote frame by checking the TXRQST and DIR bits.



1611    if((!(ui32MsgCtrl & CAN_IF2MCTL_TXRQST) && (ui32ArbReg & CAN_IF2ARB_DIR)) ||



1612       ((ui32MsgCtrl & CAN_IF2MCTL_TXRQST) && (!(ui32ArbReg & CAN_IF2ARB_DIR))))



1613    {



1614        pMsgObject->ui32Flags |= MSG_OBJ_REMOTE_FRAME;



1615    }



1616



1617    // Get the identifier out of the register, the format depends on size of



1618    // the mask.



1619    if(ui32ArbReg & CAN_IF2ARB_XTD)



1620    {



1621        // Set the 29 bit version of the Identifier for this message object.



1622        pMsgObject->ui32MsgID = ui32ArbReg & CAN_IF2ARB_ID_M;



1623



1624        pMsgObject->ui32Flags |= MSG_OBJ_EXTENDED_ID;



1625    }



1626    else



1627    {



1628        // The Identifier is an 11 bit value.



1629        pMsgObject->ui32MsgID = (ui32ArbReg &



1630                                 CAN_IF2ARB_STD_ID_M) >> CAN_IF2ARB_STD_ID_S;



1631    }



1632



1633    // Indicate that we lost some data.



1634    if(ui32MsgCtrl & CAN_IF2MCTL_MSGLST)



1635    {



1636        pMsgObject->ui32Flags |= MSG_OBJ_DATA_LOST;



1637    }



1638



1639    // Set the flag to indicate if ID masking was used.



1640    if(ui32MsgCtrl & CAN_IF2MCTL_UMASK)



1641    {



1642        if(ui32ArbReg & CAN_IF2ARB_XTD)



1643        {



1644            // The Identifier Mask is assumed to also be a 29 bit value.



1645            pMsgObject->ui32MsgIDMask = (ui32MaskReg & CAN_IF2MSK_MSK_M);



1646



1647            // If this is a fully specified Mask and a remote frame then don't



1648            // set the MSG_OBJ_USE_ID_FILTER because the ID was not really



1649            // filtered.



1650            if((pMsgObject->ui32MsgIDMask != 0x1fffffff) ||



1651               ((pMsgObject->ui32Flags & MSG_OBJ_REMOTE_FRAME) == 0))



1652            {



1653                pMsgObject->ui32Flags |= MSG_OBJ_USE_ID_FILTER;



1654            }



1655        }



1656        else



1657        {



1658            // The Identifier Mask is assumed to also be an 11 bit value.



1659            pMsgObject->ui32MsgIDMask = ((ui32MaskReg & CAN_IF2MSK_MSK_M) >>



1660                                         18);



1661



1662            // If this is a fully specified Mask and a remote frame then don't



1663            // set the MSG_OBJ_USE_ID_FILTER because the ID was not really



1664            // filtered.



1665            if((pMsgObject->ui32MsgIDMask != 0x7ff) ||



1666               ((pMsgObject->ui32Flags & MSG_OBJ_REMOTE_FRAME) == 0))



1667            {



1668                pMsgObject->ui32Flags |= MSG_OBJ_USE_ID_FILTER;



1669            }



1670        }



1671



1672        // Indicate if the extended bit was used in filtering.



1673        if(ui32MaskReg & CAN_IF2MSK_MXTD)



1674        {



1675            pMsgObject->ui32Flags |= MSG_OBJ_USE_EXT_FILTER;



1676        }



1677



1678        // Indicate if direction filtering was enabled.



1679        if(ui32MaskReg & CAN_IF2MSK_MDIR)



1680        {



1681            pMsgObject->ui32Flags |= MSG_OBJ_USE_DIR_FILTER;



1682        }



1683    }



1684



1685    // Set the interrupt flags.



1686    if(ui32MsgCtrl & CAN_IF2MCTL_TXIE)



1687    {



1688        pMsgObject->ui32Flags |= MSG_OBJ_TX_INT_ENABLE;



1689    }



1690    if(ui32MsgCtrl & CAN_IF2MCTL_RXIE)



1691    {



1692        pMsgObject->ui32Flags |= MSG_OBJ_RX_INT_ENABLE;



1693    }



1694



1695    // See if there is new data available.



1696    if(ui32MsgCtrl & CAN_IF2MCTL_NEWDAT)



1697    {



1698        // Get the amount of data needed to be read.



1699        pMsgObject->ui32MsgLen = (ui32MsgCtrl & CAN_IF2MCTL_DLC_M);



1700



1701        // Don't read any data for a remote frame, there is nothing valid in



1702        // that buffer anyway.



1703        if((pMsgObject->ui32Flags & MSG_OBJ_REMOTE_FRAME) == 0)



1704        {



1705            // Read out the data from the CAN registers.



1706            CANDataRegRead(pMsgObject->pucMsgData,



1707                           (uint32_t *)(ui32Base + CAN_O_IF2DATA),



1708                           pMsgObject->ui32MsgLen);



1709        }



1710



1711        // Now clear out the new data flag.



1712        HWREGH(ui32Base + CAN_O_IF2CMD + 2) = CAN_IF2CMD_TXRQST >> 16;



1713



1714        // Transfer the message object to the message object specified by



1715        // ui32ObjID.



1716        HWREGH(ui32Base + CAN_O_IF2CMD) = ui32ObjID & CAN_IF2CMD_MSG_NUM_M;



1717



1718        // Wait for busy bit to clear



1719        while(HWREGH(ui32Base + CAN_O_IF2CMD) & CAN_IF2CMD_BUSY)



1720        {



1721        }



1722



1723        // Indicate that there is new data in this message.



1724        pMsgObject->ui32Flags |= MSG_OBJ_NEW_DATA;



1725    }



1726    else



1727    {



1728        // Along with the MSG_OBJ_NEW_DATA not being set the amount of data



1729        // needs to be set to zero if none was available.



1730        pMsgObject->ui32MsgLen = 0;



1731    }



1732}

1737//!



1738//! \param ui32Base is the base address of the CAN controller.



1739//! \param ui32ObjID is the message object number to disable (1-32).
