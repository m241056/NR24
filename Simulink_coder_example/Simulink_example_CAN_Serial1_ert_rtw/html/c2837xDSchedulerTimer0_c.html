18      #endif



19      #ifdef PIEMASK1

21      #endif

24      #endif

28      #ifdef PIEMASK4

30      #endif



31      #ifdef PIEMASK5



32        volatile unsigned int PIEIER6_stack_save  = PieCtrlRegs.PIEIER6.all;

35        volatile unsigned int PIEIER7_stack_save  = PieCtrlRegs.PIEIER7.all;

147void configureTimer0(float baseRate, float systemClock)



148{



149  #ifndef TIC2000_SOC_BASEPRODUCT



150    #ifdef CPU1



151        EINT;



152        uint32_t ipcCountVar = 0;



153        if (0x425 == mwInitStart)



154        {



155            /* Wait until completion of CPU2 GPIO initialization */



156            while ((0x36E != mwInitEnd))



157            {



158                ipcCountVar++;



159            }



160        }



161        DINT;



162    #else



163	      uint32_t *pulMsgRam;



164	      pulMsgRam = (void *)CPU01_TO_CPU02_PASSMSG;



165        IPCLiteLtoRDataWrite(IPC_FLAG0, pulMsgRam[2], 0x36E,



166                         IPC_LENGTH_32_BITS, IPC_FLAG31);



167    #endif



168  #endif



169    



170    /* InitCpuTimers() - CPU Timers are initialized in 



171     * MW_c28xx_board.c in the generated code.



172     */



173



174	/* Configure CPU-Timer 0 to interrupt every base rate. */



175	/* Parameters:  Timer Pointer, CPU Freq in MHz, Period in usec. */



176	ConfigCpuTimer(&CpuTimer0, systemClock, baseRate * 1000000);



177	StartCpuTimer0();



178



179    #ifdef TIC2000_SOC_BASEPRODUCT



180	PieCtrlRegs.PIEIFR1.bit.INTx7 = 1;   /* Set Pending interrupt for TINT0. TINT0 will trigger as soon as the its interrupt is enabled*/



181    #endif



182    



183	EALLOW;



184	PieVectTable.TIMER0_INT = &TINT0_isr;     /* Hook interrupt to the ISR*/



185	EDIS;



186



187	PieCtrlRegs.PIEIER1.bit.INTx7 = 1;   /* Enable interrupt TINT0 */



188	IER |= M_INT1;



189}

191void stopTimer0(void)



192{

194}



195void enableTimer0Interrupt(void)



196{



197	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



198	EALLOW;



199	PieCtrlRegs.PIEIER1.all |= (1 << 6); /* Enable PIER bit for timer 0 */



200	EDIS;



201	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



202}



203



204bool disableTimer0Interrupt(void)



205{



206    bool isrEnabled = false;



207	



208    asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



209    



210    isrEnabled = PieCtrlRegs.PIEIER1.all & (1 << 6) ;



211	EALLOW;



212	PieCtrlRegs.PIEIER1.all &= ~(1 << 6); /* Disable PIER bit for timer 0 */



213	asm(" RPT #5 || NOP");



214	IFR &= 0xFFFE;



215	PieCtrlRegs.PIEACK.all = 0x1;



216	EDIS;



217	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



218    



219    return isrEnabled;



220}



221



222void restoreTimer0Interrupt(bool isrStatus){



223    if(isrStatus)



224        enableTimer0Interrupt();



225    else



226        disableTimer0Interrupt();



227}
