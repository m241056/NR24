18void init_SCI(void)



19{                                      /* initialize SCI & FIFO registers */



20  EALLOW;



21



22  /*



23   * Initialize SCI_A with following parameters:



24   *    BaudRate              : 12500000



25   *    CharacterLengthBits   : 8



26   *    EnableLoopBack        : 0



27   *    NumberOfStopBits      : 1



28   *    ParityMode            : None



29   *    SuspensionMode        : Free_run



30   *    CommMode              : Raw_data



31   */



32  CpuSysRegs.PCLKCR7.bit.SCI_A = 1;



33  asm(" NOP");



34  SciaRegs.SCICCR.bit.STOPBITS = 0;



35                    /*Number of stop bits. (0: One stop bit, 1: Two stop bits)*/



36  SciaRegs.SCICCR.bit.PARITY = 0;/*Parity mode (0: Odd parity, 1: Even parity)*/



37  SciaRegs.SCICCR.bit.PARITYENA = 0;   /*Enable Pary Mode */



38  SciaRegs.SCICCR.bit.LOOPBKENA = 0;   /*Loop Back enable*/



39  SciaRegs.SCICCR.bit.ADDRIDLE_MODE = 0;/*ADDR/IDLE Mode control*/



40  SciaRegs.SCICCR.bit.SCICHAR = 7;     /*Character length*/



41  SciaRegs.SCICTL1.bit.RXERRINTENA = 0;/*Disable receive error interrupt*/



42  SciaRegs.SCICTL1.bit.SWRESET = 1;    /*Software reset*/



43  SciaRegs.SCICTL1.bit.TXENA = 1;      /* SCI transmitter enable*/



44  SciaRegs.SCICTL1.bit.RXENA = 1;      /* SCI receiver enable*/



45  SciaRegs.SCIHBAUD.bit.BAUD = 0U;



46  SciaRegs.SCILBAUD.bit.BAUD = 1U;



47



48  /*Free run, continue SCI operation regardless of suspend*/



49  SciaRegs.SCIPRI.bit.FREESOFT = 3;



50  SciaRegs.SCIFFCT.bit.ABDCLR = 0;



51  SciaRegs.SCIFFCT.bit.CDC = 0;



52  SciaRegs.SCIFFTX.bit.SCIRST = 1;     /* SCI reset rx/tx channels*/



53  SciaRegs.SCIFFTX.bit.SCIFFENA = 1;   /* SCI FIFO enhancements are enabled.*/



54  SciaRegs.SCIFFTX.bit.TXFIFORESET = 1;/* Re-enable transmit FIFO operation.*/



55  SciaRegs.SCIFFRX.bit.RXFIFORESET = 1;/* Re-enable receive FIFO operation.*/



56  EDIS;



57}

59void init_SCI_GPIO(void)



60{



61  EALLOW;



62  GpioCtrlRegs.GPBQSEL1.bit.GPIO43 = 3;/*Asynch input GPIO43 SCIRXDA*/



63  GpioCtrlRegs.GPBPUD.bit.GPIO43 = 0;  /*Enable pull-up for GPIO43*/



64  GpioCtrlRegs.GPBGMUX1.bit.GPIO43 = 3;



65  GpioCtrlRegs.GPBMUX1.bit.GPIO43 = 3; /*Configure GPIO43 as SCIRXDA*/



66  GpioCtrlRegs.GPBPUD.bit.GPIO42 = 0;  /*Enable pull-up for GPIO42*/



67  GpioCtrlRegs.GPBGMUX1.bit.GPIO42 = 3;



68  GpioCtrlRegs.GPBMUX1.bit.GPIO42 = 3; /*Configure GPIO42 as SCITXDA*/



69  EDIS;



70}
